
with account_balance as (
    select
        customer_id,
        sum(balance) as total_balance
    from accounts
    group by customer_id
),
txn_summary as (
    select 
        a.customer_id,
        count(distinct t.transaction_id) as total_transactions,
        round(sum(t.amount),2) as total_transaction_amount,
        max(t.transaction_date) as last_txn_date
    from accounts a
    join transactions t on a.account_id = t.account_id
    group by a.customer_id
),
summary as (
    select 
        c.customer_id,
        c.full_name,
        c.gender,
        coalesce(ab.total_balance, 0) as total_balance,
        coalesce(ts.total_transactions, 0) as total_transactions,
        coalesce(ts.total_transaction_amount, 0) as total_transaction_amount,
        ts.last_txn_date
    from customers c
    left join account_balance ab on c.customer_id = ab.customer_id
    left join txn_summary ts on c.customer_id = ts.customer_id
),
     recency as(
       select*, datediff(curdate(), last_txn_date) as recency
        from summary
        ),
       segments as ( 
        select*,
       case 
       when total_balance>500000 and recency < 200  then "Premium Customers"
       when last_txn_date is null or recency > 200 then 'Dormant'
       when total_balance<10000 and total_transactions<2 then "At Risk"
       else "Mass"
       end as customer_segment
       from recency 
       )
       select* from segments;
